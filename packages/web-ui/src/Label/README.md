### Basic Usage
```jsx
<div>
  <Label>
    Email: <Input type="email" placeholder="me@example.com" />
  </Label>
  
  <Label>
    <Input type="checkbox" defaultChecked /> Checkbox
  </Label>
  
  <Label>
    <Input name="example" type="radio" defaultChecked /> Radio
  </Label>
  
  <Label>
    <Input name="example" type="radio" /> Radio
  </Label>
</div>
```
