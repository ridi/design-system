### Basic Usage
```jsx
<div>
  <Input placeholder="Enter Something" />
  <Input value="This is read only" readOnly />
  <Input value="This is disabled" disabled />
</div>
```

### Radio
```jsx
<div>
  <Input name="example1" type="radio" />
  <Input name="example1" type="radio" defaultChecked />
  <Input name="example2" type="radio" disabled />
  <Input name="example2" type="radio" disabled defaultChecked />
</div>
```
